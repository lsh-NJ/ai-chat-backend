# 聊天接口

> 前提：在 .env 中添加模型信息并在 congfig.py 中更新设置获取

1. 创建大模型服务文件：

   `touch app/services/llm_service.py`：

   ```python
   import httpxa
   from fastapi import HTT	PException

   from app.core.config import settings


   def call_llm(message: str) -> str:
       if not settings.deepseek_api_key:
           raise HTTPException(
               status_code=500,
               detail="DeepSeek API key is not configured",
           )
   	
       url = f"{settings.deepseek_base_url}/chat/completions"

   	# 使用 Bearer Token 认证，并声明请求体为 JSON
       headers = {
           "Authorization": f"Bearer {settings.deepseek_api_key}",
           "Content-Type": "application/json",
       }

       payload = {
           "model": settings.deepseek_model,
           "messages": [
               {
                   "role": "system",
                   "content": "你是一个简洁、友好、可靠的 AI 助手。",
               },
               {
                   "role": "user",
                   "content": message,
               },
           ],
           "temperature": 0.7,
       }

       try:
           with httpx.Client(
               proxy=settings.proxy_url, # 删掉取消网络代理
               timeout=60.0,
           ) as client:
               response = client.post(
                   url,
                   headers=headers,
                   json=payload,
               )

           response.raise_for_status()
           data = response.json() # 解析 json 响应体

   		# OpenAI 标准响应格式提取第一条 choice 中的消息内容
           return data["choices"][0]["message"]["content"]

       except httpx.TimeoutException as e:
           raise HTTPException(
               status_code=504,
               detail="LLM request timeout",
           ) from e

       except httpx.HTTPStatusError as e:
           raise HTTPException(
               status_code=e.response.status_code,
               detail=f"LLM API returned bad status code: {e.response.text[:300]}",
           ) from e

       except httpx.RequestError as e:
           raise HTTPException(
               status_code=502,
               detail=f"LLM API request failed: {type(e).__name__}",
           ) from e

       except KeyError as e:
           raise HTTPException(
               status_code=502,
               detail="Unexpected LLM API response format",
           ) from e
   ```
2. 修改`app/schemas/chat.py`：

   ```python
   from pydantic import BaseModel, Field

   class ChatRequest(BaseModel):
       message: str = Field(..., min_length=1, max_length=2000)
   	# 未做会话和数据库，先不管 user_id

   class ChatResponse(BaseModel):
       reply: str
   ```
3. 在 `main.py`​ 加正式 `/chat` 接口：

   ```python
   from app.services.llm_service import call_llm

   @app.post("/chat", response_model=ChatResponse)
   def chat(request: ChatRequest):
       reply = call_llm(request.message)

       return ChatResponse(reply=reply)
   ```
